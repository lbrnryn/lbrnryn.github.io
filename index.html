Simple CRUD App

Ang article na ito ay magtuturo ng simpleng CRUD(Create, Read, Update, Delete).

Tools na gagamitin: Atom Text Editor (Emmet, Atom-Live-Server), Bootstrap

Step 3 - Create index.js and add this form function

      1   const form = document.querySelector("form");
      2   form.addEventListener("submit", addName);
      3
      4   function addName(e) {
      5     const name = document.getElementById("name");
      6     const li = document.createElement("li");
      7     li.className = "list-group-item d-flex justify-content-between delete-item"
      8     li.appendChild(document.createTextNode(name.value));
      9     const a = document.createElement("a");
      10    a.innerHTML = '';
      11    li.appendChild(a);
      12    const listGroup = document.querySelector(".list-group");
      13    listGroup.appendChild(li);
      14    addToStorage(name.value);
      15    name.value = "";
      16    e.preventDefault();
      17  }
Step 4 - Add this addToStorage function
      19  function addToStorage(name) {
      20    let names;
      21    if(localStorage.getItem("names") === null) {
      22      names = [];
      23    } else {
      24      names = JSON.parse(localStorage.getItem("names"));
      25    }
      26    names.push(name);
      27
      28    localStorage.setItem("names", JSON.stringify(names));
      29  }

Step 6 - Add this removeFromStorage function
      50  const listGroup = document.querySelector(".list-group");
      51
      52  listGroup.addEventListener("click", (e) => {
      53    if (e.target.parentElement.parentElement.classList.contains("delete-item")) {
      54      e.target.parentElement.parentElement.remove();
      55   removeFromStorage(e.target.parentElement.parentElement.innerText)
      56    }
      57  })
      58
      59  function removeFromStorage(innerText) {
      60    const names = JSON.parse(localStorage.getItem("names"));
      61    names.forEach((name, index) => {
      62      if (innerText === name) {
      63        names.splice(index, 1)
      64        localStorage.setItem("names", JSON.stringify(names));
      65      }
      66    });
      67  }
